<?php
$str =<<<STR
ABCDEFGHI
STR;

preg_match("/(.*)(DEF)(.*)/", $str, $matches);

/* 出力結果
Array
(
    [0] => ABCDEFGHI
    [1] => ABC // マッチした部分の左側
    [2] => DEF // マッチした部分

)
*/
?>
